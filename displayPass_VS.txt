#version 300 es
//Display Pass
layout (location = 0) in vec4 aPosition; 
layout (location = 2) in vec2 aTexcoord; 

uniform mat4 uModelMat, uViewMat, uProjMat;

out vec4 vTexcoord;

void main()
{
	vec4 clipspace = uProjMat * uViewMat * uModelMat * aPosition;
	vTexcoord = clipspace;
	
	gl_Position = aPosition;
}
