//Takes in two Vec2 and determines where which color should go on screen
vec4 grid(in vec2 fragCoord, in vec2 resolution)
{
    vec2 modRes = resolution / (1.0, 2.0);
    
    if(fragCoord.x < modRes.x)
    {
     	return vec4(0.0, 0.0, 0.0, 0.0); 
    }
    else if(fragCoord.x > modRes.x)
    {
        return vec4(1.0, 1.0, 1.0, 1.0);
    }  
}
