// mainImage: process the current pixel (exactly one call per pixel)
//    fragColor: output final color for current pixel
//    fragCoord: input location of current pixel in image (in pixels)
void mainImage(out color4 fragColor, in sCoord fragCoord)
{
    // setup
    // test UV for input image
    sCoord uv = fragCoord / iChannelResolution[0].xy;
    
    // TESTING
    // set iChannel0 to 'Misc/Buffer A' and fetch sample
    vec4 buffA = texture(iChannel1, uv);
    vec4 buffC = texture(iChannel0, uv);
    
    //Add
    fragColor = buffA + buffC;
    
    //Mix
    fragColor = vec4(mix(buffA, buffC, 0.1));
    
    //Screen : based on https://en.wikipedia.org/wiki/Blend_modes#Screen
    fragColor = vec4(1.0) - (vec4(1.0) - buffA) * (vec4(1.0) - buffC);
    
    //Lighten Only : inspired by Lighten Only from https://en.wikipedia.org/wiki/Blend_modes#Screen
    fragColor = max(buffA, buffC);
}
