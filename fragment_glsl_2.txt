#version 300 es
//Screen Pass
#ifdef GL_ES
precision highp float;
#endif //GL_ES

layout (location = 0) out vec4 rtFragColor;

in vec4 vNormal;
in vec4 vTexcoord;

uniform sampler2D uTex;

void main()
{
	int option = 0;
	if(option == 0)
	{
		rtFragColor = vNormal;
	}
	else if(option == 1)
	{
		vec2 uv = vTexcoord.xy;
		vec4 color = texture(uTex, uv);
		rtFragColor = color;
	}
	else
	{
		//vec3 pos = gSphere2.position;
		//rtFragColor = vNormal;
	}

	
}
