#version 300 es
//Luminance Pass
//Ian Melvin
layout (location = 0) in vec4 aPosition; 
layout (location = 2) in vec4 aTexcoord; 

uniform mat4 uModelMat, uViewMat, uProjMat;

out vec4 vTexcoord;

//Passes aTexcoord to the PS with varying
void main()
{
	//vTexcoord = aTexcoord;
	vTexcoord = (aPosition / aPosition.w) * 0.5 + 0.5;
	
	gl_Position = aPosition / aPosition.w;
}
