#version 300 es
//Display Pass
//Ian Melvin
#ifdef GL_ES
precision highp float;
#endif //GL_ES

layout (location = 0) out vec4 rtFragColor;

in vec4 vTexcoord;

uniform sampler2D uTex, uTex2;

//Outputs the results of previous passes
void main()
{
	vec4 ndc = vTexcoord / vTexcoord.w;
	vec2 uv = ndc.xy * 0.5 + 0.5;
	vec4 color = texture(uTex, uv);
		
	vec4 color2 = texture(uTex2, uv);

	rtFragColor = color + color2;
}
